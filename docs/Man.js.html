/**
 * Create a new Man
 * @class 
 * @classdesc 一个描述人的类，用于创建后续的实例
 */
var Man =  {
  /**
   * 说话
   * @return {string} [返回]
   */
  say: function(){
    return ",hello"
  }
}

 * @return {string}      [组合成我的名字字符串]
 */

  return 'My name is ' + name
}
